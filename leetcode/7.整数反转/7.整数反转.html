# 7.整数反转

/*
 * @lc app=leetcode.cn id=7 lang=javascript
 *
 * [7] 整数反转
 */
/**
 * @param {number} x
 * @return {number}
 */
var reverse = function(x) {
    var flag = x > 0 ? true: false,
        str = Math.abs(x) + '';

    var resStr = str.split('').reverse().join('') - 0;
    res = flag ? resStr : resStr * -1

    if (res <= (Math.pow(2, 31) * -1) || res >= ( Math.pow(2, 31) - 1 )) {
        return 0;
    }

    return res;
};
